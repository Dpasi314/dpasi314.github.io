def createGraph(p, lower=False,upper=False,two=False,alpha=0.5,shade=True):
    x = np.linspace(stats.norm.ppf(0.001), stats.norm.ppf(0.999), 100) # Compute standard normal curve
    plt.plot(x, stats.norm.pdf(x)) # Plot the standard normal
    aval = 0
    if(shade == True): aval = 0.4
    if(lower):
        plt.axvspan(stats.norm.ppf(0.001),stats.norm.ppf(alpha),alpha=0.4)
        plt.axvspan(stats.norm.ppf(alpha),stats.norm.ppf(p),alpha=aval,color="lightgreen")
        plt.axvline(stats.norm.ppf(p),color="red")
        plt.show()
        return;
    if(upper):
        plt.axvspan(stats.norm.ppf(1-alpha),stats.norm.ppf(0.9999),alpha=0.4)
        plt.axvspan(stats.norm.ppf(p),stats.norm.ppf(1-alpha),alpha=aval,color="lightgreen")
        plt.axvline(stats.norm.ppf(p),color="red")
        plt.show()
        return;
    if(two):
        plt.axvspan(stats.norm.ppf(0.975),stats.norm.ppf(0.9999),alpha=0.4)
        plt.axvspan(stats.norm.ppf(0.0001),stats.norm.ppf(0.025),alpha=0.4)
        plt.axvspan(stats.norm.ppf(p),stats.norm.ppf(0.975),alpha=aval,color="lightgreen")
        plt.axvline(stats.norm.ppf(p),color="red")
        plt.show()
        return;
